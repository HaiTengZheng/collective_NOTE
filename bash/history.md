# history
```bash
# list all previous commands
history

# clear the history
history -c

# expand to line n of bash history
!n

# expand to last command
!!

# expand to last command containing "text"
!?text

# expand to command n lines ago
!-n

# expand to last command with first occurrence of "foo" replaced by "bar"
^foo^bar^

# expand to the current command
!#
```
